CREATE DATABASE `UUSERS`;
USE `UUSERS`;
SET NAMES 'utf8mb4';
SET CHARSET 'utf8mb4';

CREATE TABLE USSERS(
    `id` INT AUTO_INCREMENT NOT NULL,
    `name` VARCHAR(64) NOT NULL,
    `email` VARCHAR(64) NOT NULL,
    `password` VARCHAR(50) NOT NULL,
    UNIQUE (`name`),
    PRIMARY KEY (`id`)
) ENGINE = INNODB;
